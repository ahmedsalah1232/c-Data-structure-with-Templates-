#include <iostream>

using namespace std;
template<class T>
struct node
{
    T item;
    node<T>* next;
    node<T>* previous;
};
template<class T>
class DoubleLinkedlist
{
    node<T> *first,*last;
    int length;
public:
    DoubleLinkedlist();
    void InsertAtHead(const T data);
    void InsertAtTail(const T);
    void InsertAt(int position,const T);
    void RemoveAtTail();
    void RemoveAtHead();
    void RemoveAt(int index);
    void RepalceAt(const T,int index);
    bool IsExist(T element);
    bool IsItemAtEqual(const T,int index);
    bool IsEmpty();
    int LinkedListSize();
    void Clear();
    void Print();

};
template<class T>
DoubleLinkedlist<T>::DoubleLinkedlist()
{
    first=last=NULL;
    length=0;
}
template<class T>
void DoubleLinkedlist<T>::InsertAtHead(const T data)
{
    node<T>* newnode=new node<T>;
    newnode->item=data;
    if(length==0)
    {
        newnode->next=newnode->previous=NULL;
        first=newnode;
        last=newnode;
    }
    else
    {
        newnode->next=first;
        newnode->previous=NULL;
        first->previous=newnode;
        first=newnode;
    }
    length++;

}
template<class T>
void DoubleLinkedlist<T>::InsertAtTail(const T data)
{
    node<T>* newnode=new node<T>;
    newnode->item=data;
    if(length==0)
    {
        first=newnode;
        last=newnode;
        newnode->previous=NULL;
        newnode->next=NULL;
    }
    else
    {
        newnode->next=NULL;
        newnode->previous=last;
        last->next=newnode;
        last=newnode;
    }
    length++;
}
template<class T>
void DoubleLinkedlist<T>::InsertAt(int position,const T data)
{
    node<T>* current=first;
    node<T>* newnode=new node<T>;
    newnode->item=data;
    if(position<0||position>length)
    {
        cout<<"The position is out of the range"<<endl;
    }
    else if(position==0)
    {
        InsertAtHead(data);
    }
    else if(position==length)
    {
        InsertAtTail(data);
    }
    else
    {
        for(int i=1;i<position;i++)
        {
            current=current->next;
        }
        newnode->next=current->next;
        newnode->previous=current;
        current->next->previous=newnode;
        current->next=newnode;
        length++;
    }
}
template<class T>
void DoubleLinkedlist<T>::RemoveAtHead()
{
    if(length==0)
    {
        cout<<"ERROR:empty list"<<endl;
    }
    else if(length==1)
    {
        delete first;
        first=last=NULL;
        length=0;
    }
    else
    {
        node<T>* temp=first;
        first=first->next;
        first->previous=NULL;
        delete temp;
        length--;
    }
}
template<class T>
void DoubleLinkedlist<T>::RemoveAtTail()
{
    if(length==0)
    {
        cout<<"ERROR:empty list"<<endl;
    }
    else if(length==1)
    {
        delete first;
        first=last=NULL;
        length=0;
    }
    else
    {
        node<T> *current=last;
        last=last->previous;
        last->next=NULL;
        delete current;
        length--;
    }
}
template<class T>
void DoubleLinkedlist<T>::RemoveAt(int index)
{
    if(index<0||index>=length)
    {
        cout<<"The index is out of the range"<<endl;
    }
    else if(index==0)
    {
        RemoveAtHead();
    }
    else if(index==length-1)
    {
        RemoveAtTail();
    }
    else
    {
        node<T> *current=first->next;
        for(int i=1;i<index;i++)
        {
            current=current->next;
        }
        current->previous->next=current->next;
        current->next->previous=current->previous;
        delete current;
        length--;
    }
}
template<class T>
void DoubleLinkedlist<T>::RepalceAt(T data,int index)
{
    if(index<0||index>length)
    {
        cout<<"The index is out of the range"<<endl;
    }
    else{
    node<T> *current=first;
    for(int i=1;i<index;i++)
    {
        current=current->next;
    }
    current->item=data;
    }
}
template<class T>
bool DoubleLinkedlist<T>::IsExist(T element)
{
    node<T> *current=first;
    for(int i=1;i<=length;i++)
    {
        if(current->item==element)
        {
            return true;
            break;
        }
        current=current->next;
    }
    return false;
}
template<class T>
bool DoubleLinkedlist<T>::IsItemAtEqual(T element,int index)
{
    node<T> *current=first;
    for(int i=1;i<=index;i++)
    {
        current=current->next;
    }
    if(current->item==element)
        return true;
    else
        return false;
}
template<class T>
bool DoubleLinkedlist<T>::IsEmpty()
{
    return length==0;
}
template<class T>
int DoubleLinkedlist<T>::LinkedListSize()
{
    return length;
}
template<class T>
void DoubleLinkedlist<T>::Clear()
{
    while(length>0)
    {
        RemoveAtHead();
        length--;
    }
}
template<class T>
void DoubleLinkedlist<T>::Print()
{
    if(IsEmpty())
    {
        cout<<"The list is Empty"<<endl;
    }
    else{
    node<T> *current=first;
    while(current!=NULL)
    {
        cout<<current->item<<" ";
        current=current->next;
    }
    }
    cout<<endl;
}
int main()
{
    cout<<5<<endl;
    DoubleLinkedlist<int>numbers;
    numbers.InsertAtHead(10);
    numbers.InsertAtHead(30);
    numbers.InsertAtHead(60);
    numbers.InsertAtHead(80);
    numbers.InsertAtHead(12);
    numbers.RemoveAt(0);
    numbers.RepalceAt(16,3);
    numbers.RemoveAtTail();
    numbers.Print();
    return 0;
}
